import java.io.IOException;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) throws IOException {
        Scanner scanner = new Scanner(System.in);
        int a = scanner.nextInt();
        int b = scanner.nextInt();
        int count = 0;

        for(int i=1; i<=a; i++){
            if(a%i == 0){
                count++;
                if(count==b){
                    System.out.print(i);
                }
            }
        }
        if(count < b){
            System.out.print(0);
        }
    }
}
